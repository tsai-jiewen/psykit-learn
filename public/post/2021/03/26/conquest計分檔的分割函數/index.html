item_split <- function(dat, n) {
  require(tidyverse)
  require(stringr)
  l <- list()
  for (i in 1:n) {
    l[[i]] <- dat %>% str_sub(i, i) %>% as.numeric()
  }
  i = i + 1
  
  df <- matrix(unlist(l), nrow = length(l), byrow = T) %>% t() %>% as_tibble()
  df
}
